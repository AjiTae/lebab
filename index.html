xto6 is a JavaScript transpiler

It helps you to keep your JavaScript codes up-to-date !

Transpile ES5 code into ES6 live !

var someClass = function () {
    console.log('This is the constructor.');
};

someClass.prototype.someOuterMethod = someFunction;

someClass.prototype.someInnerMethod = function (s, t) {
    var result = s + t;
    console.log(result);
    return result;
};

Object.defineProperty(someClass.prototype, 'someAccessor', {
    get: function () {
      return this._some;
    },
    set: function (value) {
      this._some = value;
    }
});

function someFunction(a, b) {
    return a + b;

Usage


        

            xto6 transpiles your ES5 code to ES6. It works like other available transpilers (like babel
            and traceur) but
            in reverse order!
        


        
Install it using npm :

        
$ npm install -g xto6


        
Transpile your old-fashioned code using the xto6 cli tool.


        
xto6 es5.js -o es6.js
